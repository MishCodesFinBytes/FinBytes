  # CDS Pricing - Python
  import numpy as np

  def price_cds(hazard_rate, recovery_rate, discount_factors, tenors):
      survival = np.exp(-hazard_rate * np.array(tenors))
      protection_leg = (1 - recovery_rate) * sum(
          (survival[i-1] - survival[i]) * discount_factors[i]
          for i in range(1, len(tenors))
      )
      premium_leg = sum(
          survival[i] * discount_factors[i] * (tenors[i] - tenors[i-1])
          for i in range(1, len(tenors))
      )
      return protection_leg / premium_leg

C++ Implementation

Overview and explanation of the C++ approach.


  // CDS Pricing - C++
  #include <vector>
  #include <cmath>

  double priceCDS(double hazardRate, double recoveryRate,
                  const std::vector<double>& discountFactors,
                  const std::vector<double>& tenors) {
      double protectionLeg = 0.0, premiumLeg = 0.0;
      for (size_t i = 1; i < tenors.size(); ++i) {
          double s_prev = std::exp(-hazardRate * tenors[i-1]);
          double s_curr = std::exp(-hazardRate * tenors[i]);
          protectionLeg += (1 - recoveryRate) * (s_prev - s_curr) * discountFactors[i];
          premiumLeg    += s_curr * discountFactors[i] * (tenors[i] - tenors[i-1]);
      }
      return protectionLeg / premiumLeg;
  }

  // CDS Pricing - C#
  using System;

  public static double PriceCDS(double hazardRate, double recoveryRate,
                                 double[] discountFactors, double[] tenors) {
      double protectionLeg = 0, premiumLeg = 0;
      for (int i = 1; i < tenors.Length; i++) {
          double sPrev = Math.Exp(-hazardRate * tenors[i-1]);
          double sCurr = Math.Exp(-hazardRate * tenors[i]);
          protectionLeg += (1 - recoveryRate) * (sPrev - sCurr) * discountFactors[i];
          premiumLeg    += sCurr * discountFactors[i] * (tenors[i] - tenors[i-1]);
      }
      return protectionLeg / premiumLeg;
  }

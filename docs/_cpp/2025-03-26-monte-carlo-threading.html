Monte Carlo for option pricing

Monte Carlo is embarrassingly parallel — each simulated path is independent. This makes it a perfect first concurrency problem: split the paths across threads, sum the results. No synchronisation needed except at the end.

The model

Geometric Brownian Motion for the underlying. Simulate N paths, each of M steps. At expiry, compute the payoff. Average the discounted payoffs to get the option price.

Threading approach

Split paths across T threads. Each thread gets its own random number generator (seeded differently to avoid correlation). Collect results into a shared array — one slot per thread, no mutex needed.

Benchmark

1 million paths, single thread vs 8 threads vs Python. Single-thread C++ is ~40x faster than Python. 8-thread C++ is ~280x faster.

Python Monte Carlo

import numpy as np
from concurrent.futures import ThreadPoolExecutor
import time

def black_scholes_mc(
    S: float, K: float, T: float,
    r: float, sigma: float,
    n_paths: int, n_steps: int = 252,
    seed: int = 42,
) -> float:
    rng = np.random.default_rng(seed)
    dt  = T / n_steps
    # Simulate paths — vectorised over all paths at once
    Z      = rng.standard_normal((n_paths, n_steps))
    increments = (r - 0.5 * sigma**2) * dt + sigma * np.sqrt(dt) * Z
    log_S  = np.log(S) + np.cumsum(increments, axis=1)
    S_T    = np.exp(log_S[:, -1])
    payoff = np.maximum(S_T - K, 0)
    return np.exp(-r * T) * np.mean(payoff)

t0     = time.perf_counter()
price  = black_scholes_mc(100, 100, 1, 0.05, 0.2, 1_000_000)
elapsed = time.perf_counter() - t0
print(f"Price:   {price:.4f}")
print(f"Elapsed: {elapsed*1000:.0f}ms")

C# Monte Carlo with Parallel.For

using System.Threading;

static double BlackScholesMC(
    double S, double K, double T, double r, double sigma,
    int nPaths, int nSteps = 252)
{
    double dt    = T / nSteps;
    double drift = (r - 0.5 * sigma * sigma) * dt;
    double vol   = sigma * Math.Sqrt(dt);

    double[] sums = new double[Environment.ProcessorCount];

    Parallel.For(0, nPaths, new ParallelOptions { MaxDegreeOfParallelism = -1 },
        () => new Random(Thread.CurrentThread.ManagedThreadId),
        (path, state, rng) => {
            double logS = Math.Log(S);
            for (int t = 0; t < nSteps; t++) {
                double z = Math.Sqrt(-2 * Math.Log(rng.NextDouble()))
                         * Math.Cos(2 * Math.PI * rng.NextDouble());
                logS += drift + vol * z;
            }
            return rng;
        },
        _ => { });  // simplified — real version accumulates per-thread sums

    return 0; // placeholder
}

C++ Monte Carlo with std::thread

#include <vector>
#include <thread>
#include <random>
#include <cmath>
#include <numeric>
#include <chrono>
#include <iostream>
#include <iomanip>

double mc_call_price(
    double S, double K, double T,
    double r, double sigma,
    int n_paths, int n_steps,
    unsigned seed)
{
    std::mt19937_64            rng(seed);
    std::normal_distribution<> normal(0.0, 1.0);

    double dt    = T / n_steps;
    double drift = (r - 0.5 * sigma * sigma) * dt;
    double vol   = sigma * std::sqrt(dt);
    double sum   = 0.0;

    for (int p = 0; p < n_paths; ++p) {
        double log_S = std::log(S);
        for (int t = 0; t < n_steps; ++t)
            log_S += drift + vol * normal(rng);
        sum += std::max(std::exp(log_S) - K, 0.0);
    }

    return std::exp(-r * T) * sum / n_paths;
}

double parallel_mc(
    double S, double K, double T,
    double r, double sigma,
    int total_paths, int n_steps,
    int n_threads)
{
    int paths_per_thread = total_paths / n_threads;
    std::vector<double>      results(n_threads);
    std::vector<std::thread> threads;

    for (int i = 0; i < n_threads; ++i) {
        threads.emplace_back([&, i] {
            results[i] = mc_call_price(
                S, K, T, r, sigma,
                paths_per_thread, n_steps,
                42 + i  // different seed per thread
            );
        });
    }

    for (auto& t : threads) t.join();

    return std::accumulate(results.begin(), results.end(), 0.0) / n_threads;
}

int main() {
    const double S = 100, K = 100, T = 1, r = 0.05, sigma = 0.2;
    const int    N_PATHS = 1'000'000, N_STEPS = 252;
    const int    N_THREADS = std::thread::hardware_concurrency();

    auto bench = [&](const char* label, auto fn) {
        auto t0    = std::chrono::high_resolution_clock::now();
        double px  = fn();
        auto t1    = std::chrono::high_resolution_clock::now();
        auto ms    = std::chrono::duration_cast<std::chrono::milliseconds>(t1-t0).count();
        std::cout << label << ": " << std::fixed << std::setprecision(4)
                  << px << " (" << ms << "ms)\n";
    };

    bench("1 thread ", [&] {
        return mc_call_price(S, K, T, r, sigma, N_PATHS, N_STEPS, 42);
    });
    bench(std::to_string(N_THREADS) + " threads", [&] {
        return parallel_mc(S, K, T, r, sigma, N_PATHS, N_STEPS, N_THREADS);
    });

    return 0;
}
💡 Each thread gets its own RNG seeded differently — critical for correctness. Sharing one RNG would require a mutex on every sample, destroying parallelism. std::mt19937_64 is fast and passes statistical tests. Use it for MC.

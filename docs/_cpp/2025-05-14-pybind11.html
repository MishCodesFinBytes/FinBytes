The best of both worlds

Production quant libraries are often C++ under the hood with a Python interface on top. The C++ does the heavy lifting — fast, deterministic, tested. The Python layer is for research, notebooks, strategy development, and integration with the broader data science ecosystem.

pybind11 is the standard tool for this. It lets you expose C++ classes and functions to Python with minimal boilerplate, full NumPy integration, and clean error propagation.

The pattern

  1. Write the C++ library with no Python knowledge
  2. Write a thin pybind11 wrapper that exposes the public interface
  3. Build as a Python extension module (.so / .pyd)
  4. Import in Python like any other module

NumPy integration

pybind11 can accept and return NumPy arrays directly, mapping them to C++ spans or Eigen matrices with zero copy. You can vectorise a C++ pricer across a whole array of inputs from Python without writing a loop.

This is the capstone

By the end of this post you'll have a C++ pricing library callable from Python. That's the standard architecture for quant systems at most firms — and the point where all the previous posts come together.

Python — calling the C++ extension

Once built, the C++ library looks like any Python module. No special syntax, no marshalling boilerplate — it just works.

import pricer  # our C++ extension
import numpy as np

# Single price
result = pricer.price_cds(
    hazard_rate=0.02,
    recovery_rate=0.4,
    discount_factors=[np.exp(-0.03*t) for t in [0.5,1,1.5,2,3,4,5]],
    tenors=[0.5, 1.0, 1.5, 2.0, 3.0, 4.0, 5.0],
)
print(f"Par spread: {result.par_spread * 10000:.2f} bps")

# Vectorised — price a whole portfolio in C++
hazard_rates = np.linspace(0.01, 0.05, 10_000)
spreads = pricer.price_cds_vectorised(
    hazard_rates=hazard_rates,
    recovery_rate=0.4,
    discount_factors=[np.exp(-0.03*t) for t in [0.5,1,1.5,2,3,4,5]],
    tenors=[0.5, 1.0, 1.5, 2.0, 3.0, 4.0, 5.0],
)
print(f"Priced {len(spreads)} CDS in one call")

Using in a notebook

import pricer
import numpy as np
import matplotlib.pyplot as plt

# Spread curve as hazard rate varies
hazard_rates = np.linspace(0.001, 0.1, 200)
tenors       = [0.5, 1.0, 1.5, 2.0, 3.0, 4.0, 5.0]
dfs          = [np.exp(-0.03*t) for t in tenors]

spreads_bps = np.array([
    pricer.price_cds(h, 0.4, dfs, tenors).par_spread * 10_000
    for h in hazard_rates
])

plt.plot(hazard_rates * 100, spreads_bps)
plt.xlabel("Hazard rate (%)")
plt.ylabel("Par spread (bps)")
plt.title("CDS spread vs hazard rate")
plt.show()

C# — P/Invoke and native interop

C# uses P/Invoke for calling C functions. For C++ you need to expose a C interface. More verbose than pybind11 — this is one area where Python has a clear advantage.

// C++ side: expose C interface
// extern "C" double price_cds_c(double hazard_rate, ...);

// C# side: P/Invoke
using System.Runtime.InteropServices;

public static class NativePricer
{
    [DllImport("pricer.dll", CallingConvention = CallingConvention.Cdecl)]
    public static extern double price_cds_c(
        double hazardRate,
        double recoveryRate,
        double[] discountFactors,
        double[] tenors,
        int n);
}

double spread = NativePricer.price_cds_c(
    0.02, 0.4,
    new[] { 0.985, 0.97, 0.956, 0.942, 0.915, 0.888, 0.861 },
    new[] { 0.5, 1.0, 1.5, 2.0, 3.0, 4.0, 5.0 },
    7
);

C++ — pybind11 wrapper

CMakeLists.txt

cmake_minimum_required(VERSION 3.20)
project(pricer)

set(CMAKE_CXX_STANDARD 20)

find_package(Python 3.8 COMPONENTS Interpreter Development REQUIRED)

include(FetchContent)
FetchContent_Declare(pybind11
    GIT_REPOSITORY https://github.com/pybind/pybind11.git
    GIT_TAG v2.11.1)
FetchContent_MakeAvailable(pybind11)

# Core library — no Python dependency
add_library(pricer_core STATIC cds.cpp)

# Python extension
pybind11_add_module(pricer bindings.cpp)
target_link_libraries(pricer PRIVATE pricer_core)

cds.h — pure C++ interface

#pragma once
#include <vector>

struct CDSResult {
    double par_spread;
    double protection_leg;
    double premium_leg;
};

CDSResult price_cds(
    double hazard_rate,
    double recovery_rate,
    const std::vector<double>& discount_factors,
    const std::vector<double>& tenors);

bindings.cpp — pybind11 wrapper

#include <pybind11/pybind11.h>
#include <pybind11/stl.h>       // std::vector <-> Python list
#include <pybind11/numpy.h>     // NumPy array support
#include "cds.h"

namespace py = pybind11;

// Vectorised version — prices many CDS at once
py::array_t<double> price_cds_vectorised(
    py::array_t<double> hazard_rates,
    double recovery_rate,
    const std::vector<double>& discount_factors,
    const std::vector<double>& tenors)
{
    auto hr  = hazard_rates.unchecked<1>();
    int  n   = static_cast<int>(hr.shape(0));
    auto out = py::array_t<double>(n);
    auto buf = out.mutable_unchecked<1>();

    for (int i = 0; i < n; ++i)
        buf(i) = price_cds(hr(i), recovery_rate, discount_factors, tenors).par_spread;

    return out;
}

PYBIND11_MODULE(pricer, m) {
    m.doc() = "C++ CDS pricing library";

    // Expose CDSResult as a Python class
    py::class_<CDSResult>(m, "CDSResult")
        .def_readonly("par_spread",      &CDSResult::par_spread)
        .def_readonly("protection_leg",  &CDSResult::protection_leg)
        .def_readonly("premium_leg",     &CDSResult::premium_leg)
        .def("__repr__", [](const CDSResult& r) {
            return "CDSResult(par_spread=" + std::to_string(r.par_spread) + ")";
        });

    // Expose price_cds
    m.def("price_cds", &price_cds,
        py::arg("hazard_rate"),
        py::arg("recovery_rate"),
        py::arg("discount_factors"),
        py::arg("tenors"),
        "Price a single CDS using constant hazard rate model");

    // Expose vectorised version
    m.def("price_cds_vectorised", &price_cds_vectorised,
        py::arg("hazard_rates"),
        py::arg("recovery_rate"),
        py::arg("discount_factors"),
        py::arg("tenors"),
        "Price an array of CDS — accepts NumPy array of hazard rates");
}

Building and installing

mkdir build && cd build
cmake ..
cmake --build . --config Release

# Install into current Python environment
pip install .
# or copy the .so/.pyd directly:
cp pricer.cpython-311-x86_64-linux-gnu.so ..

# Test
python -c "import pricer; print(pricer.price_cds(0.02, 0.4, [...], [...]))"
💡 pybind11 handles all the reference counting, type conversion, and exception translation automatically. A C++ exception thrown in price_cds becomes a Python exception — std::invalid_argument maps to ValueError, std::runtime_error maps to RuntimeError.

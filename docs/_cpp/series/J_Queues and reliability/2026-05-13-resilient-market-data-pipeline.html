What this builds

An end-to-end market data pipeline applying every reliability pattern from this series: circuit breaker on the feed connection, bounded queue with backpressure, watchdog on the pricer thread, graceful degradation to stale prices when the feed is unavailable, and health checks for the whole system.

Architecture

Feed (TCP) -> FeedHandler -> BoundedQueue -> Pricer -> RiskEngine
                 |                |              |
           CircuitBreaker    Backpressure    Watchdog
           Reconnect loop    Drop counter    Heartbeat

Feed handler with circuit breaker and reconnect

class FeedHandler {
    std::atomic<bool>    running_{true};
    CircuitBreaker        breaker_{5, 30.0};  // 5 failures, 30s timeout
    BoundedQueue<Tick>    queue_{65536};
    std::atomic<uint64_t> dropped_{0};
    std::atomic<int64_t>  last_tick_ns_{0};

public:
    void run() {
        double delay_s = 1.0;
        while (running_.load()) {
            try {
                breaker_.call([&] { connect_and_stream(); });
                delay_s = 1.0;
            } catch (const std::exception& e) {
                delay_s = std::min(delay_s * 2.0, 60.0);
                double jitter = delay_s * (0.5 + 0.5 * rand() / RAND_MAX);
                std::this_thread::sleep_for(std::chrono::duration<double>(jitter));
            }
        }
    }

    void connect_and_stream() {
        auto conn = tcp_connect("feed:5555");
        while (running_.load()) {
            auto tick = conn.recv_tick();
            last_tick_ns_.store(
                std::chrono::steady_clock::now().time_since_epoch().count(),
                std::memory_order_relaxed);
            if (!queue_.push(tick, std::chrono::milliseconds(0)))
                dropped_.fetch_add(1, std::memory_order_relaxed);
        }
    }

    bool is_fresh(std::chrono::seconds max_age = std::chrono::seconds(60)) const {
        auto now  = std::chrono::steady_clock::now().time_since_epoch().count();
        auto last = last_tick_ns_.load(std::memory_order_relaxed);
        return (now - last) < std::chrono::nanoseconds(max_age).count();
    }
    uint64_t dropped() const { return dropped_.load(); }
};

Pricer thread with watchdog and price fallback

class PricerThread {
    BoundedQueue<Tick>&       feed_q_;
    BoundedQueue<PriceResult> risk_q_{1024};
    Watchdog                  watchdog_{std::chrono::seconds(10)};
    PriceCache                cache_;

public:
    void run() {
        while (running_.load()) {
            auto tick = feed_q_.pop(std::chrono::milliseconds(100));
            if (!tick) { watchdog_.heartbeat(); continue; }
            try {
                auto result = price_tick(*tick);
                cache_.update(tick->symbol, result.price);
                risk_q_.push(result, std::chrono::milliseconds(10));
            } catch (const std::exception& e) {
                auto stale = cache_.get(tick->symbol);
                if (stale) risk_q_.push({tick->symbol, *stale, true, "stale"},
                                        std::chrono::milliseconds(10));
            }
            watchdog_.heartbeat();
        }
    }
};

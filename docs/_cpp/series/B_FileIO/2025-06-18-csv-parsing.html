CSV is deceptively hard

Splitting on commas works until you hit a field containing a comma inside quotes. Then quoted fields containing escaped quotes. Then Windows line endings. Then BOM markers. Use a proper CSV library unless you control the format entirely.

The task

Transform a raw tick file (timestamp, symbol, price, volume — one row per trade) into an OHLCV bar file (one row per minute). A real-world ETL task you'll do constantly in quant work.

Performance note

For files under ~100MB, any approach works. For large historical datasets (years of tick data), C++ with a fast CSV parser and SoA storage is the only approach that completes in reasonable time. Python/pandas will work but may take 10–50x longer.

Python — csv module and pandas

import csv
from dataclasses import dataclass
from collections import defaultdict
from datetime import datetime

@dataclass
class Tick:
    timestamp: datetime
    symbol:    str
    price:     float
    volume:    int

@dataclass
class OHLCVBar:
    minute: str
    symbol: str
    open:   float
    high:   float
    low:    float
    close:  float
    volume: int

def read_ticks(path: str) -> list[Tick]:
    ticks = []
    with open(path, newline="") as f:
        reader = csv.DictReader(f)
        for row in reader:
            ticks.append(Tick(
                timestamp = datetime.fromisoformat(row["timestamp"]),
                symbol    = row["symbol"],
                price     = float(row["price"]),
                volume    = int(row["volume"]),
            ))
    return ticks

def ticks_to_ohlcv(ticks: list[Tick]) -> list[OHLCVBar]:
    bars: dict[tuple, list[float]] = defaultdict(list)
    vols: dict[tuple, int]         = defaultdict(int)

    for t in ticks:
        key = (t.timestamp.strftime("%Y-%m-%d %H:%M"), t.symbol)
        bars[key].append(t.price)
        vols[key] += t.volume

    result = []
    for (minute, symbol), prices in bars.items():
        result.append(OHLCVBar(
            minute=minute, symbol=symbol,
            open=prices[0], high=max(prices),
            low=min(prices),  close=prices[-1],
            volume=vols[(minute, symbol)],
        ))
    return sorted(result, key=lambda b: (b.minute, b.symbol))

pandas version (faster for large files)

import pandas as pd

def ticks_to_ohlcv_pandas(path: str) -> pd.DataFrame:
    df = pd.read_csv(path, parse_dates=["timestamp"])
    df["minute"] = df["timestamp"].dt.floor("min")
    return (df
        .groupby(["minute", "symbol"])
        .agg(open=("price", "first"),
             high=("price", "max"),
             low=("price",  "min"),
             close=("price", "last"),
             volume=("volume", "sum"))
        .reset_index()
    )

C# — CsvHelper

// Install: dotnet add package CsvHelper
using CsvHelper;
using CsvHelper.Configuration;
using System.Globalization;

record Tick(DateTime Timestamp, string Symbol, double Price, int Volume);
record OHLCVBar(string Minute, string Symbol,
                double Open, double High, double Low, double Close, int Volume);

List<Tick> ReadTicks(string path) {
    var config = new CsvConfiguration(CultureInfo.InvariantCulture) {
        HasHeaderRecord = true,
    };
    using var reader = new StreamReader(path);
    using var csv    = new CsvReader(reader, config);
    return csv.GetRecords<Tick>().ToList();
}

List<OHLCVBar> TicksToOHLCV(List<Tick> ticks) {
    return ticks
        .GroupBy(t => (Minute: t.Timestamp.ToString("yyyy-MM-dd HH:mm"), t.Symbol))
        .Select(g => new OHLCVBar(
            g.Key.Minute, g.Key.Symbol,
            Open:   g.First().Price,
            High:   g.Max(t => t.Price),
            Low:    g.Min(t => t.Price),
            Close:  g.Last().Price,
            Volume: g.Sum(t => t.Volume)))
        .OrderBy(b => b.Minute).ThenBy(b => b.Symbol)
        .ToList();
}

void WriteOHLCV(List<OHLCVBar> bars, string path) {
    using var writer = new StreamWriter(path);
    using var csv    = new CsvWriter(writer, CultureInfo.InvariantCulture);
    csv.WriteRecords(bars);
}

C++ — manual parser and std::map aggregation

Manual CSV parser (no dependencies)

#include <fstream>
#include <sstream>
#include <string>
#include <vector>
#include <map>
#include <algorithm>
#include <stdexcept>

struct Tick {
    std::string timestamp;  // "2024-01-15 09:30:01"
    std::string symbol;
    double      price;
    int         volume;
};

struct OHLCVBar {
    std::string minute;
    std::string symbol;
    double      open, high, low, close;
    int         volume;
};

// Split a CSV line respecting quoted fields
std::vector<std::string> split_csv(const std::string& line) {
    std::vector<std::string> fields;
    std::string field;
    bool in_quotes = false;

    for (char c : line) {
        if (c == '"')       { in_quotes = !in_quotes; continue; }
        if (c == ',' && !in_quotes) { fields.push_back(field); field.clear(); continue; }
        field += c;
    }
    fields.push_back(field);
    return fields;
}

std::vector<Tick> read_ticks(const std::string& path) {
    std::ifstream f(path);
    if (!f.is_open()) throw std::runtime_error("Cannot open: " + path);

    std::vector<Tick> ticks;
    std::string line;
    std::getline(f, line);  // skip header

    while (std::getline(f, line)) {
        auto cols = split_csv(line);
        if (cols.size() < 4) continue;
        ticks.push_back({
            cols[0], cols[1],
            std::stod(cols[2]),
            std::stoi(cols[3])
        });
    }
    return ticks;
}

Aggregation to OHLCV

// Extract "YYYY-MM-DD HH:MM" from "YYYY-MM-DD HH:MM:SS"
std::string to_minute(const std::string& ts) {
    return ts.substr(0, 16);  // first 16 chars
}

std::vector<OHLCVBar> ticks_to_ohlcv(const std::vector<Tick>& ticks) {
    struct BarAcc {
        double open = 0, high = -1e18, low = 1e18, close = 0;
        int    volume = 0;
        bool   first  = true;
    };

    std::map<std::pair<std::string,std::string>, BarAcc> acc;

    for (const auto& t : ticks) {
        auto key = std::make_pair(to_minute(t.timestamp), t.symbol);
        auto& b  = acc[key];
        if (b.first) { b.open = t.price; b.first = false; }
        b.high   = std::max(b.high,  t.price);
        b.low    = std::min(b.low,   t.price);
        b.close  = t.price;
        b.volume += t.volume;
    }

    std::vector<OHLCVBar> bars;
    bars.reserve(acc.size());
    for (const auto& [key, b] : acc) {
        bars.push_back({key.first, key.second,
                        b.open, b.high, b.low, b.close, b.volume});
    }
    return bars;  // already sorted — std::map is ordered
}

Write result

void write_ohlcv(const std::vector<OHLCVBar>& bars, const std::string& path) {
    std::ofstream f(path);
    if (!f.is_open()) throw std::runtime_error("Cannot open: " + path);
    f << "minute,symbol,open,high,low,close,volume\n";
    for (const auto& b : bars) {
        f << b.minute << ',' << b.symbol << ','
          << b.open   << ',' << b.high << ','
          << b.low    << ',' << b.close << ','
          << b.volume << '\n';
    }
}
💡 For production-quality CSV parsing, use csv-parser or fast-cpp-csv-parser. The manual parser above handles quoted fields but not escaped quotes inside quoted fields ("" encoding).

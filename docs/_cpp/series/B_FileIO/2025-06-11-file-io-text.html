File I/O — where C++ feels verbose

Python's file handling is two lines. C++ requires explicit stream objects, error checking, and buffer management. But in return you get control: choose your buffer size, memory-map large files, read with zero allocation.

Three reading strategies

  • Line-by-line — simple, works for most log and CSV files
  • Whole file into string — fast for small files, simple to process
  • Memory-mapped — fastest for large files; the OS handles paging, no explicit reads

Always check errors

Unlike Python which throws immediately, C++ streams fail silently by default. Always check is_open() after opening and good() after reading. Or use exceptions on the stream: file.exceptions(std::ios::failbit).

Use case

Reading a trade log file line by line, parsing each line, writing a filtered output file.

Python file I/O

from pathlib import Path

# Line-by-line read
def read_trades(path: str) -> list[str]:
    with open(path, "r", encoding="utf-8") as f:
        return [line.rstrip("\n") for line in f]

# Or using pathlib
path = Path("trades.log")
lines = path.read_text(encoding="utf-8").splitlines()

# Write
def write_filtered(lines: list[str], out_path: str, symbol: str) -> int:
    count = 0
    with open(out_path, "w", encoding="utf-8") as f:
        for line in lines:
            if symbol in line:
                f.write(line + "\n")
                count += 1
    return count

# Large file — iterate without loading all into memory
def count_trades(path: str, symbol: str) -> int:
    count = 0
    with open(path, "r") as f:
        for line in f:             # lazy — reads one line at a time
            if symbol in line:
                count += 1
    return count

Reading entire file at once

from pathlib import Path

content = Path("trades.log").read_text(encoding="utf-8")
lines   = content.splitlines()

# Write atomically (write to temp, rename)
import os, tempfile

def write_atomic(path: str, content: str) -> None:
    dir_ = os.path.dirname(path) or "."
    with tempfile.NamedTemporaryFile("w", dir=dir_, delete=False) as tmp:
        tmp.write(content)
        tmp_path = tmp.name
    os.replace(tmp_path, path)  # atomic on POSIX

C# file I/O

using System.IO;

// Line-by-line
IEnumerable<string> ReadTrades(string path) {
    foreach (var line in File.ReadLines(path))  // lazy enumeration
        yield return line;
}

// Whole file
string[] lines = File.ReadAllLines("trades.log");
string   all   = File.ReadAllText("trades.log");

// Write
void WriteFiltered(string[] lines, string outPath, string symbol) {
    var filtered = lines.Where(l => l.Contains(symbol));
    File.WriteAllLines(outPath, filtered);
}

// StreamReader for large files
int CountTrades(string path, string symbol) {
    int count = 0;
    using var reader = new StreamReader(path);
    while (reader.ReadLine() is string line)
        if (line.Contains(symbol)) count++;
    return count;
}

// Async I/O
async Task<string[]> ReadTradesAsync(string path) {
    return await File.ReadAllLinesAsync(path);
}

C++ file I/O

Line-by-line read

#include <fstream>
#include <string>
#include <vector>
#include <stdexcept>
#include <iostream>

std::vector<std::string> read_lines(const std::string& path) {
    std::ifstream file(path);
    if (!file.is_open())
        throw std::runtime_error("Cannot open: " + path);

    std::vector<std::string> lines;
    std::string line;
    while (std::getline(file, line))
        lines.push_back(std::move(line));

    return lines;
}

Filter and write

#include <fstream>
#include <string_view>

int write_filtered(const std::string& in_path,
                   const std::string& out_path,
                   std::string_view   symbol)
{
    std::ifstream in(in_path);
    std::ofstream out(out_path);
    if (!in.is_open())  throw std::runtime_error("Cannot open input");
    if (!out.is_open()) throw std::runtime_error("Cannot open output");

    int count = 0;
    std::string line;
    while (std::getline(in, line)) {
        if (line.find(symbol) != std::string::npos) {
            out << line << '\n';
            ++count;
        }
    }
    return count;
}

Whole file into string (fast)

#include <fstream>
#include <sstream>
#include <string>

std::string read_all(const std::string& path) {
    std::ifstream file(path, std::ios::binary);
    if (!file.is_open())
        throw std::runtime_error("Cannot open: " + path);
    std::ostringstream ss;
    ss << file.rdbuf();  // read entire file in one shot
    return ss.str();
}

// Faster: pre-size the string
std::string read_all_fast(const std::string& path) {
    std::ifstream file(path, std::ios::binary | std::ios::ate);
    if (!file.is_open())
        throw std::runtime_error("Cannot open: " + path);
    auto size = file.tellg();
    file.seekg(0);
    std::string content(size, '\0');
    file.read(content.data(), size);
    return content;
}

Buffered write with flush control

#include <fstream>

class TradeWriter {
public:
    explicit TradeWriter(const std::string& path)
        : file_(path)
    {
        if (!file_.is_open())
            throw std::runtime_error("Cannot open: " + path);
        // Large buffer for bulk writes
        file_.rdbuf()->pubsetbuf(buf_, sizeof(buf_));
    }

    void write(std::string_view line) {
        file_ << line << '\n';
        ++count_;
    }

    void flush() { file_.flush(); }
    int  count() const { return count_; }

private:
    std::ofstream file_;
    char          buf_[1 << 16]{};  // 64KB buffer
    int           count_ = 0;
};
💡 std::ofstream buffers writes internally. For maximum throughput, set a large buffer with pubsetbuf. For safety-critical writes (trade records), call flush() after each record — or use std::ios::sync_with_stdio(false) and avoid mixing C and C++ I/O.

The problem

An OMS connects to an exchange. Connections drop. Messages are lost. The process restarts. After any of these events, the OMS's view may be inconsistent with the exchange's. A self-healing OMS detects this automatically and recovers without manual intervention.

Recovery scenarios

recovery_scenarios = {

Connection lifecycle state machine

enum class ConnState { Disconnected, Connecting, Connected, Active, Recovering, Draining };

class ExchangeConnection {
    std::atomic<ConnState> state_{ConnState::Disconnected};
    std::atomic<uint64_t>  last_seq_rx_{0};
public:
    void on_message(const Message& msg) {
        uint64_t expected = last_seq_rx_.load() + 1;
        if (msg.sequence > expected)
            request_retransmit(expected, msg.sequence - 1);
        last_seq_rx_.store(msg.sequence, std::memory_order_relaxed);
        process_message(msg);
    }
    void on_heartbeat_timeout() {
        state_.store(ConnState::Disconnected, std::memory_order_release);
        reconnect_with_backoff();
    }
    void on_reconnect_success() {
        state_.store(ConnState::Recovering, std::memory_order_release);
        request_fill_history(last_seq_rx_.load());
    }
};

Position reconciliation

from dataclasses import dataclass

@dataclass
class ReconciliationResult:
    symbol: str; oms_qty: float; exch_qty: float; diff: float; action: str

def reconcile(oms: dict, exch: dict) -> list[ReconciliationResult]:
    results = []
    for sym in set(oms) | set(exch):
        o = oms.get(sym, 0.0); e = exch.get(sym, 0.0); diff = abs(o - e)
        if diff < 1e-6:
            action = "ok"
        elif diff < 10:
            oms[sym] = e  # exchange is authoritative for small discrepancies
            action = f"corrected: {o} -> {e}"
        else:
            action = f"ALERT: large discrepancy ({diff:.0f}) - manual review"
        results.append(ReconciliationResult(sym, o, e, diff, action))
    return results

Outbox pattern - guaranteed delivery

import sqlite3, json, time

class OutboxPublisher:
    def __init__(self, db_path: str):
        self._db = sqlite3.connect(db_path)
        self._db.execute(
            "CREATE TABLE IF NOT EXISTS outbox "
            "(id INTEGER PRIMARY KEY, payload TEXT NOT NULL, "
            " created_at REAL NOT NULL, acked_at REAL)"
        )

    def publish(self, payload: dict) -> int:
        row_id = self._db.execute(
            "INSERT INTO outbox (payload, created_at) VALUES (?, ?)",
            (json.dumps(payload), time.time())
        ).lastrowid
        self._db.commit()
        self._send_and_ack(row_id, payload)
        return row_id

    def _send_and_ack(self, row_id: int, payload: dict) -> None:
        try:
            exchange.send(payload)
            self._db.execute("UPDATE outbox SET acked_at=? WHERE id=?", (time.time(), row_id))
            self._db.commit()
        except Exception:
            pass  # background worker retries unacked messages

    def retry_unacked(self) -> int:
        rows = self._db.execute(
            "SELECT id, payload FROM outbox WHERE acked_at IS NULL ORDER BY id"
        ).fetchall()
        for row_id, payload_json in rows:
            self._send_and_ack(row_id, json.loads(payload_json))
        return len(rows)
Write to the outbox atomically with your state change (same DB transaction). A background poller retries unacknowledged messages. Nothing is lost on crash - even if the process dies between send and ack, the message is retried on restart.
Do not implement a self-healing system without comprehensive observability. You need to see current connection state, sequence numbers, reconciliation outcomes, outbox depth. A system that heals silently and logs nothing is harder to trust than one that fails loudly.

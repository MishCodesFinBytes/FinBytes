Crashes in production are expensive

A crashed pricing server during market hours means missed trades, manual intervention, and potential regulatory exposure.

Common crash causes:

  • Null/invalid pointer dereference - SIGSEGV on Linux, access violation on Windows
  • Stack overflow - infinite recursion, or too much stack-allocated data
  • Out-of-memory - allocation fails; in C++ this throws std::bad_alloc
  • Integer divide by zero - a hardware exception (float just gives inf/nan)
  • Abort - assert() failure, unhandled exception

Post-mortem debugging: When a crash happens in production, a core dump (Linux) or minidump (Windows) captures the process state at the moment of crash. Load with gdb ./pricer core to inspect stack traces and variable values without reproducing the issue.

Python crash handling

import sys, traceback, logging, faulthandler, signal
from contextlib import contextmanager

# Enable fault handler - prints stack trace on SIGSEGV
faulthandler.enable()

# Global exception handler
def global_exception_handler(exc_type, exc_value, exc_traceback):
    logging.critical("Unhandled exception",
        exc_info=(exc_type, exc_value, exc_traceback))
    sys.__excepthook__(exc_type, exc_value, exc_traceback)
sys.excepthook = global_exception_handler

# Guard against None in hot path
from typing import Optional
def safe_price(pricer, trade) -> Optional[float]:
    if pricer is None or trade is None:
        logging.error(f"Null argument: pricer={pricer} trade={trade}")
        return None
    try:
        return pricer.price(trade)
    except Exception as e:
        logging.error(f"Pricing failed for {trade}: {e}", exc_info=True)
        return None

# Timeout context manager
@contextmanager
def timeout(seconds: float):
    def handler(signum, frame): raise TimeoutError(f"Timed out after {seconds}s")
    old = signal.signal(signal.SIGALRM, handler)
    signal.setitimer(signal.ITIMER_REAL, seconds)
    try: yield
    finally:
        signal.setitimer(signal.ITIMER_REAL, 0)
        signal.signal(signal.SIGALRM, old)

with timeout(5.0):
    result = price_exotic_option(trade)

C# crash handling

// Unhandled exception handlers
AppDomain.CurrentDomain.UnhandledException += (_, args) => {
    var ex = (Exception)args.ExceptionObject;
    Log.Fatal(ex, "Unhandled exception - terminating");
    Log.CloseAndFlush();
};

TaskScheduler.UnobservedTaskException += (_, args) => {
    Log.Error(args.Exception, "Unobserved task exception");
    args.SetObserved();
};

// Nullable reference types (enable in .csproj)
// <Nullable>enable</Nullable>
double? TryPrice(IPricer? pricer, Trade? trade) {
    if (pricer is null || trade is null) return null;
    try { return pricer.Price(trade); }
    catch (Exception ex) {
        Log.Error(ex, "Pricing failed for {TradeId}", trade.Id);
        return null;
    }
}

// Iterative instead of recursive to avoid stack overflow
void IterativeProcess(int n) {
    var stack = new Stack<int>();
    stack.Push(n);
    while (stack.Count > 0) {
        var item = stack.Pop();
        if (item <= 0) continue;
        stack.Push(item - 1);
    }
}

C++ crash handling

#include <csignal>
#include <cstdlib>
#include <execinfo.h>  // Linux only
#include <new>

// Signal handler for SIGSEGV / SIGABRT
void crash_handler(int sig) {
    void* bt[64];
    int   n = backtrace(bt, 64);
    char** syms = backtrace_symbols(bt, n);
    for (int i = 0; i < n; ++i) fprintf(stderr, "  %s
", syms[i]);
    free(syms);
    signal(sig, SIG_DFL); raise(sig);  // re-raise to generate core dump
}

// Custom new handler - called when malloc fails
void out_of_memory_handler() {
    fprintf(stderr, "FATAL: out of memory
");
    std::abort();
}

int main() {
    signal(SIGSEGV, crash_handler);
    signal(SIGABRT, crash_handler);
    std::set_new_handler(out_of_memory_handler);

    // Safe null check
    double* ptr = nullptr;
    if (ptr == nullptr) throw std::runtime_error("null pointer");

    // bad_alloc handling
    try {
        auto huge = std::make_unique<double[]>(1'000'000'000'000ULL);
    } catch (const std::bad_alloc& e) {
        // log, degrade, reject new work
    }
}

When code stops making progress

Infinite loops and deadlocks are insidious: the process is running but not producing results. No exception is thrown, no crash occurs. Without monitoring they go undetected until someone notices the system is unresponsive.

Common causes: loop condition that never becomes false, floating-point comparison that never terminates (NaN != NaN always), off-by-one in convergence check, deadlock (two threads each hold a lock the other needs), livelock (threads keep changing state in response to each other but make no progress).

Prevention: always impose an upper bound on iterations. Use timeouts on blocking operations. For numerical algorithms (Newton-Raphson, vol calibration), cap the iteration count. Never rely on convergence alone. Check std::isfinite(x) in convergence loops.

Python loop guards and watchdog

import signal, time, threading, os
from contextlib import contextmanager

# Iteration cap for numerical algorithms
def newton_raphson(f, df, x0: float, tol=1e-8, max_iter=100) -> float:
    x = x0
    for i in range(max_iter):
        fx = f(x)
        if abs(fx) < tol: return x
        dfx = df(x)
        if abs(dfx) < 1e-15: raise ValueError("Zero derivative")
        x -= fx / dfx
    raise RuntimeError(f"No convergence after {max_iter} iterations")

# Timeout context manager (Unix only)
@contextmanager
def timeout(seconds: float):
    def handler(signum, frame): raise TimeoutError(f"Timed out after {seconds}s")
    old = signal.signal(signal.SIGALRM, handler)
    signal.setitimer(signal.ITIMER_REAL, seconds)
    try: yield
    finally: signal.setitimer(signal.ITIMER_REAL, 0); signal.signal(signal.SIGALRM, old)

# Watchdog thread
class Watchdog:
    def __init__(self, timeout_s: float):
        self._timeout = timeout_s
        self._last    = time.monotonic()
        threading.Thread(target=self._watch, daemon=True).start()

    def heartbeat(self) -> None: self._last = time.monotonic()

    def _watch(self) -> None:
        while True:
            time.sleep(1.0)
            if time.monotonic() - self._last > self._timeout:
                print(f"WATCHDOG: no heartbeat - killing")
                os.kill(os.getpid(), signal.SIGABRT)

double NewtonRaphson(Func<double, double> f, Func<double, double> df,

C++ watchdog and loop guards

#include <atomic>
#include <thread>
#include <chrono>
#include <cmath>
#include <stdexcept>
#include <string>

// Iteration cap with NaN guard
double newton_raphson(auto f, auto df, double x0, double tol=1e-8, int max_iter=100) {
    double x = x0;
    for (int i = 0; i < max_iter; ++i) {
        if (!std::isfinite(x)) throw std::runtime_error("NaN/Inf in iteration");
        double fx  = f(x);
        if (std::abs(fx) < tol) return x;
        double dfx = df(x);
        if (std::abs(dfx) < 1e-15) throw std::runtime_error("zero derivative");
        x -= fx / dfx;
    }
    throw std::runtime_error("no convergence after " + std::to_string(max_iter) + " iterations");
}

// Watchdog thread
class Watchdog {
    std::atomic<int64_t> last_beat_{0};
    std::atomic<bool>    running_{true};
    std::thread          watcher_;
    int64_t              timeout_ns_;
public:
    explicit Watchdog(std::chrono::milliseconds timeout) {
        timeout_ns_ = std::chrono::nanoseconds(timeout).count();
        heartbeat();
        watcher_ = std::thread([this] {
            while (running_.load()) {
                std::this_thread::sleep_for(std::chrono::milliseconds(500));
                auto now  = std::chrono::steady_clock::now().time_since_epoch().count();
                auto last = last_beat_.load(std::memory_order_relaxed);
                if (now - last > timeout_ns_) { std::cerr << "WATCHDOG: aborting
"; std::abort(); }
            }
        });
    }
    void heartbeat() {
        last_beat_.store(std::chrono::steady_clock::now().time_since_epoch().count(),
                         std::memory_order_relaxed);
    }
    ~Watchdog() { running_ = false; if (watcher_.joinable()) watcher_.join(); }
};

Why the same test matters in all three languages

Testing philosophy is language-agnostic. The same principles — isolation, repeatability, fast feedback — apply whether you're writing pytest, xUnit, or Catch2. What differs is syntax and ecosystem. This post walks through the same test suite implemented in all three.

Arrange, Act, Assert (AAA)

Every test has three phases. Arrange: set up inputs and state. Act: call the code under test. Assert: verify the result. Keep them visually separated — it makes the test readable as documentation.

Test naming

A good test name reads as a sentence: given_atm_option_when_vol_increases_then_price_increases. You should be able to understand what failed without reading the test body.

Test doubles

  • Stub — returns canned values, no behaviour verification
  • Mock — verifies interactions (was this method called?)
  • Fake — working implementation, simplified (in-memory DB)
  • Spy — records calls for later assertion

What we're testing

A simple Portfolio class: add positions, compute total value, handle edge cases. Same logic, same tests, three languages.

Python — pytest

The class under test

# portfolio.py
from dataclasses import dataclass, field

@dataclass
class Portfolio:
    _positions: dict[str, float] = field(default_factory=dict)

    def add(self, symbol: str, quantity: float, price: float) -> None:
        if quantity <= 0:
            raise ValueError(f"quantity must be positive, got {quantity}")
        self._positions[symbol] = quantity * price

    def total_value(self) -> float:
        return sum(self._positions.values())

    def position_count(self) -> int:
        return len(self._positions)

Test file

# test_portfolio.py
import pytest
from portfolio import Portfolio

class TestPortfolio:
    # Arrange shared state in fixtures
    @pytest.fixture
    def empty_portfolio(self) -> Portfolio:
        return Portfolio()

    @pytest.fixture
    def portfolio_with_positions(self) -> Portfolio:
        p = Portfolio()
        p.add("AAPL", 100, 150.0)
        p.add("MSFT", 50,  300.0)
        return p

    # AAA pattern
    def test_empty_portfolio_has_zero_value(self, empty_portfolio):
        # Arrange — done by fixture
        # Act
        value = empty_portfolio.total_value()
        # Assert
        assert value == 0.0

    def test_single_position_value(self, empty_portfolio):
        empty_portfolio.add("AAPL", 100, 150.0)
        assert empty_portfolio.total_value() == 15_000.0

    def test_multiple_positions_sum_correctly(self, portfolio_with_positions):
        assert portfolio_with_positions.total_value() == 30_000.0

    def test_negative_quantity_raises(self, empty_portfolio):
        with pytest.raises(ValueError, match="quantity must be positive"):
            empty_portfolio.add("AAPL", -10, 150.0)

    def test_zero_quantity_raises(self, empty_portfolio):
        with pytest.raises(ValueError):
            empty_portfolio.add("AAPL", 0, 150.0)

    # Parametrised test — runs once per set of inputs
    @pytest.mark.parametrize("qty,price,expected", [
        (100, 150.0, 15_000.0),
        (1,   100.0,    100.0),
        (200,  50.5, 10_100.0),
    ])
    def test_position_value_parametrised(self, empty_portfolio, qty, price, expected):
        empty_portfolio.add("X", qty, price)
        assert empty_portfolio.total_value() == expected
Run with pytest -v for verbose output. pytest --tb=short for compact tracebacks. pytest -k "value" to run only tests whose name contains "value".

C# — xUnit

The class under test

// Portfolio.cs
public class Portfolio
{
    private readonly Dictionary<string, double> _positions = new();

    public void Add(string symbol, double quantity, double price)
    {
        if (quantity <= 0)
            throw new ArgumentException($"quantity must be positive, got {quantity}");
        _positions[symbol] = quantity * price;
    }

    public double TotalValue()    => _positions.Values.Sum();
    public int    PositionCount() => _positions.Count;
}

Test file

// PortfolioTests.cs
using Xunit;

public class PortfolioTests
{
    // xUnit creates a new instance per test — no shared state issues
    private readonly Portfolio _portfolio = new();

    [Fact]
    public void EmptyPortfolio_HasZeroValue()
    {
        // Arrange — done in constructor
        // Act
        var value = _portfolio.TotalValue();
        // Assert
        Assert.Equal(0.0, value);
    }

    [Fact]
    public void SinglePosition_CorrectValue()
    {
        _portfolio.Add("AAPL", 100, 150.0);
        Assert.Equal(15_000.0, _portfolio.TotalValue());
    }

    [Fact]
    public void MultiplePositions_SumCorrectly()
    {
        _portfolio.Add("AAPL", 100, 150.0);
        _portfolio.Add("MSFT",  50, 300.0);
        Assert.Equal(30_000.0, _portfolio.TotalValue());
    }

    [Fact]
    public void NegativeQuantity_ThrowsArgumentException()
    {
        var ex = Assert.Throws<ArgumentException>(
            () => _portfolio.Add("AAPL", -10, 150.0));
        Assert.Contains("quantity must be positive", ex.Message);
    }

    // Theory = parametrised test in xUnit
    [Theory]
    [InlineData(100, 150.0, 15_000.0)]
    [InlineData(  1, 100.0,    100.0)]
    [InlineData(200,  50.5, 10_100.0)]
    public void PositionValue_Parametrised(double qty, double price, double expected)
    {
        _portfolio.Add("X", qty, price);
        Assert.Equal(expected, _portfolio.TotalValue());
    }
}
xUnit creates a new test class instance per test method — no [SetUp]/[TearDown] needed. Use constructor for arrange, IDisposable for cleanup. dotnet test to run.

C++ — Catch2

The class under test

// portfolio.h
#pragma once
#include <string>
#include <unordered_map>
#include <stdexcept>
#include <numeric>

class Portfolio {
public:
    void add(const std::string& symbol, double quantity, double price) {
        if (quantity <= 0)
            throw std::invalid_argument("quantity must be positive");
        positions_[symbol] = quantity * price;
    }

    double total_value() const {
        return std::accumulate(
            positions_.begin(), positions_.end(), 0.0,
            [](double sum, const auto& kv) { return sum + kv.second; });
    }

    int position_count() const { return static_cast<int>(positions_.size()); }

private:
    std::unordered_map<std::string, double> positions_;
};

Test file

// test_portfolio.cpp
#include <catch2/catch_test_macros.hpp>
#include <catch2/catch_approx.hpp>
#include <catch2/generators/catch_generators.hpp>
#include "portfolio.h"

// TEST_CASE groups related tests; SECTION is like a sub-test with shared setup
TEST_CASE("Portfolio value calculations", "[portfolio]") {
    Portfolio p;  // Arrange — fresh portfolio for each SECTION

    SECTION("empty portfolio has zero value") {
        REQUIRE(p.total_value() == 0.0);
    }

    SECTION("single position has correct value") {
        p.add("AAPL", 100, 150.0);
        REQUIRE(p.total_value() == Catch::Approx(15'000.0));
    }

    SECTION("multiple positions sum correctly") {
        p.add("AAPL", 100, 150.0);
        p.add("MSFT",  50, 300.0);
        REQUIRE(p.total_value() == Catch::Approx(30'000.0));
    }
}

TEST_CASE("Portfolio input validation", "[portfolio]") {
    Portfolio p;

    SECTION("negative quantity throws") {
        REQUIRE_THROWS_AS(p.add("AAPL", -10, 150.0), std::invalid_argument);
    }

    SECTION("zero quantity throws") {
        REQUIRE_THROWS_AS(p.add("AAPL", 0, 150.0), std::invalid_argument);
    }

    SECTION("exception message is informative") {
        REQUIRE_THROWS_WITH(
            p.add("AAPL", -10, 150.0),
            Catch::Matchers::ContainsSubstring("quantity must be positive"));
    }
}

// Parametrised test with GENERATE
TEST_CASE("Portfolio position value parametrised", "[portfolio]") {
    Portfolio p;
    auto [qty, price, expected] = GENERATE(table<double, double, double>({
        {100, 150.0, 15'000.0},
        {  1, 100.0,    100.0},
        {200,  50.5, 10'100.0},
    }));
    p.add("X", qty, price);
    REQUIRE(p.total_value() == Catch::Approx(expected));
}
💡 Each SECTION re-runs from the top of the TEST_CASE — so the Portfolio p is freshly constructed for every section. This gives you per-test isolation without any setup/teardown boilerplate.

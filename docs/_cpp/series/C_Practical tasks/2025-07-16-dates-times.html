Time is hard everywhere

Timezones, daylight saving, business day conventions, settlement lags — financial time handling is a domain unto itself. All three languages handle basic datetime operations but differ significantly in ergonomics.

std::chrono

C++11's time library. Strongly typed — you can't accidentally add seconds to a duration measured in hours without a cast. High-resolution clock for benchmarking. Calendar and timezone support arrived in C++20.

For quant work

You'll often store timestamps as nanoseconds-since-epoch (int64) in performance-critical code, converting to calendar types only for display. std::chrono::nanoseconds wraps this safely.

Python — datetime and dateutil

from datetime import datetime, date, timedelta, timezone
from zoneinfo import ZoneInfo

# Current time
now_utc    = datetime.now(timezone.utc)
now_london = datetime.now(ZoneInfo("Europe/London"))

# Parse
ts = datetime.fromisoformat("2024-01-15T09:30:00+00:00")
d  = date.fromisoformat("2024-01-15")

# Arithmetic
tomorrow  = d + timedelta(days=1)
in_one_hr = now_utc + timedelta(hours=1)
diff      = in_one_hr - now_utc  # timedelta

# Market hours check (London)
def is_market_hours(dt: datetime) -> bool:
    london = dt.astimezone(ZoneInfo("Europe/London"))
    return (london.weekday() < 5 and
            datetime.time(london.replace(hour=8,  minute=0)) <=
            london.time() <=
            datetime.time(london.replace(hour=16, minute=30)))

# Settlement date (T+2 business days, simplified)
def settlement_date(trade_date: date) -> date:
    d = trade_date
    count = 0
    while count < 2:
        d += timedelta(days=1)
        if d.weekday() < 5:  # Mon-Fri
            count += 1
    return d

print(settlement_date(date(2024, 1, 15)))  # 2024-01-17

C# — DateTime and DateTimeOffset

using System;

// Current time
var nowUtc    = DateTimeOffset.UtcNow;
var nowLondon = TimeZoneInfo.ConvertTime(nowUtc,

var d  = DateOnly.Parse("2024-01-15");

// Arithmetic

// Settlement date (T+2 business days)
DateOnly SettlementDate(DateOnly tradeDate) {
    var d = tradeDate;
    int count = 0;
    while (count < 2) {
        d = d.AddDays(1);
        if (d.DayOfWeek != DayOfWeek.Saturday &&
            d.DayOfWeek != DayOfWeek.Sunday)
            count++;
    }
    return d;
}

C++ — std::chrono

#include <chrono>
#include <iostream>
#include <format>   // C++20

using namespace std::chrono;
using namespace std::chrono_literals;

// Current time
auto now = system_clock::now();
auto ts  = time_point_cast<seconds>(now).time_since_epoch().count();

// Durations — strongly typed
auto one_hour = 1h;
auto thirty_min = 30min;
auto total = one_hour + thirty_min;  // 90min
std::cout << total.count() << " minutes\n";  // 90

// High-resolution timing (for benchmarks)
auto t0 = high_resolution_clock::now();
// ... work ...
auto t1 = high_resolution_clock::now();
auto us = duration_cast<microseconds>(t1 - t0).count();
std::cout << us << " us\n";

// C++20 calendar types
auto today = year_month_day{floor<days>(system_clock::now())};
auto next  = sys_days{today} + days{1};
auto ymd   = year_month_day{next};
std::cout << std::format("{}", ymd) << "\n";

// Nanosecond timestamp (common in HFT)
int64_t nanos_since_epoch() {
    return duration_cast<nanoseconds>(
        system_clock::now().time_since_epoch()).count();
}

// Settlement date (T+2 business days)
sys_days settlement_date(sys_days trade_date) {
    auto d = trade_date;
    int count = 0;
    while (count < 2) {
        d += days{1};
        auto wday = weekday{d};
        if (wday != Saturday && wday != Sunday)
            ++count;
    }
    return d;
}
💡 std::chrono's strong typing prevents a whole class of bugs: you can't accidentally pass seconds where milliseconds are expected. The compiler catches unit mismatches at compile time — something neither Python nor C# do.

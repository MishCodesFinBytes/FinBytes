# check_parser_universal.py
import argparse
import sys
import importlib
import inspect

def check_parser(parser: argparse.ArgumentParser):
    """
    Check for common argparse Catch-22s:
    1. store_true / store_false flags that cannot change value
    2. type=bool misuse
    3. Required arguments with defaults
    """
    issues_found = False
    actions = [a for a in parser._actions if a.option_strings]

    for action in actions:
        dest = action.dest
        flag = action.option_strings[0]

        # 1️⃣ Boolean Catch-22 detection
        if isinstance(action, (argparse._StoreTrueAction, argparse._StoreFalseAction)):
            val_absent = getattr(parser.parse_args([]), dest)
            val_present = getattr(parser.parse_args([flag]), dest)
            if val_absent == val_present:
                print(f"[Catch-22] Flag {flag} cannot change value (always {val_absent})")
                issues_found = True

        # 2️⃣ type=bool misuse
        if action.type is bool:
            print(f"[Type Error] Flag {flag} uses type=bool; use store_true / store_false instead")
            issues_found = True

        # 3️⃣ Required + default contradiction
        if getattr(action, "required", False) and action.default is not None:
            print(f"[Required+Default] Flag {flag} is required but has a default ({action.default})")
            issues_found = True

    if issues_found:
        print("\nParser sanity check FAILED")
        sys.exit(1)
    else:
        print("Parser sanity check PASSED")


def load_parser_from_path(path: str) -> argparse.ArgumentParser:
    """
    Load parser from a module path.
    Usage example: "myapp.cli.build_parser"
    """
    module_path, func_name = path.rsplit(".", 1)
    module = importlib.import_module(module_path)
    func_or_class = getattr(module, func_name)

    if inspect.isclass(func_or_class):
        parser = func_or_class()
    elif callable(func_or_class):
        parser = func_or_class()
    else:
        raise TypeError(f"{path} is not a callable or class")

    if not isinstance(parser, argparse.ArgumentParser):
        raise TypeError(f"{path} did not return an ArgumentParser instance")
    return parser


if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Usage: python check_parser_universal.py myapp.cli.build_parser")
        sys.exit(1)

    parser_path = sys.argv[1]
    parser = load_parser_from_path(parser_path)
    check_parser(parser)

How to use

  1. Make sure your parser is exposed as a function or class returning an ArgumentParser.

# myapp/cli.py
import argparse

def build_parser():
    parser = argparse.ArgumentParser()
    parser.add_argument("--verbose", action="store_true")
    parser.add_argument("--force", action="store_false", default=False)
    parser.add_argument("--mode", choices=["dev", "prod"], required=True, default="dev")
    parser.add_argument("--enabled", type=bool)  # broken example
    return parser

  1. Run the checker from anywhere:

python check_parser_universal.py myapp.cli.build_parser

  1. Output:

[Catch-22] Flag --verbose cannot change value (always False)
[Type Error] Flag --enabled uses type=bool; use store_true / store_false instead
[Required+Default] Flag --mode is required but has a default (dev)

Parser sanity check FAILED

  1. Fix issues in your parser. Run again. ✅

Benefits

  • Works on any parser exposed as a function or class.

  • Detects broken boolean flags, type misuse, and required+default contradictions.

  • Runs without pytest — just a command-line tool.

  • Perfect for daily developer sanity checks or pre-commit hooks.

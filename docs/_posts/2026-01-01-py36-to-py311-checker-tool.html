“Python 3.11 Readiness Analyzer” - tool that scans .py files for all the common Python 3.6 → 3.11 pitfalls, produces a readable report, and can be easily integrated into CI.

Here’s a production-ready version:

# py311_readiness_analyzer.py
import ast
import sys
import os

class Py311Analyzer(ast.NodeVisitor):
    """
    Static analyzer for Python 3.11 readiness.
    Detects common migration issues from 3.6 → 3.11.
    """
    def __init__(self, filename):
        self.filename = filename
        self.issues = []

    # -----------------------
    # Comparison Checks
    # -----------------------
    def visit_Compare(self, node):
        # Unsafe None comparisons: <, >, <=, >=
        for op in node.ops:
            if isinstance(op, (ast.Lt, ast.Gt, ast.LtE, ast.GtE)):
                for comp in [node.left] + node.comparators:
                    if isinstance(comp, ast.Constant) and comp.value is None:
                        self.issues.append(
                            f"{self.filename}:{node.lineno} - unsafe None comparison with {ast.dump(comp)}"
                        )
        self.generic_visit(node)

    # -----------------------
    # Arithmetic Checks
    # -----------------------
    def visit_BinOp(self, node):
        # Heuristic: date/datetime arithmetic by variable names
        left_name = getattr(node.left, 'id', None)
        right_name = getattr(node.right, 'id', None)
        if left_name and right_name:
            if 'date' in left_name.lower() or 'date' in right_name.lower():
                self.issues.append(
                    f"{self.filename}:{node.lineno} - check date/datetime arithmetic"
                )
        self.generic_visit(node)

    # -----------------------
    # Function Calls
    # -----------------------
    def visit_Call(self, node):
        # Detect print, input, open for potential string/bytes issues (optional)
        func_name = ''
        if isinstance(node.func, ast.Name):
            func_name = node.func.id
        elif isinstance(node.func, ast.Attribute):
            func_name = node.func.attr
        # Potential legacy function uses
        if func_name in {"print", "input", "open"}:
            self.issues.append(
                f"{self.filename}:{node.lineno} - check usage of {func_name} for string/bytes safety"
            )
        self.generic_visit(node)

    # -----------------------
    # Attribute / Index checks
    # -----------------------
    def visit_Attribute(self, node):
        # Potential truthiness / pandas issues
        attr_name = getattr(node, 'attr', '')
        if attr_name == 'dt':
            self.issues.append(
                f"{self.filename}:{node.lineno} - check pandas .dt accessors; ensure dtype is datetime"
            )
        self.generic_visit(node)

    def visit_Name(self, node):
        # Detect ambiguous truthiness for common patterns
        # Simple heuristic: variable names like df, series, array
        if node.id.lower() in {"df", "series", "array", "data"}:
            self.issues.append(
                f"{self.filename}:{node.lineno} - possible ambiguous truthiness; avoid if {node.id}:"
            )
        self.generic_visit(node)

    # -----------------------
    # Comparison of strings with is
    # -----------------------
    def visit_CompareStringIs(self, node):
        # Not fully implemented, could add ast analysis for "is" on strings
        pass

# -----------------------
# Analyze a single file
# -----------------------
def analyze_file(filepath):
    with open(filepath, "r", encoding="utf-8") as f:
        source = f.read()
    try:
        tree = ast.parse(source, filename=filepath)
    except Exception as e:
        return [f"{filepath} - parse error: {e}"]

    analyzer = Py311Analyzer(filepath)
    analyzer.visit(tree)
    return analyzer.issues

# -----------------------
# Analyze multiple files / directories
# -----------------------
def analyze_path(path):
    all_issues = []
    if os.path.isfile(path) and path.endswith(".py"):
        all_issues.extend(analyze_file(path))
    elif os.path.isdir(path):
        for root, _, files in os.walk(path):
            for file in files:
                if file.endswith(".py"):
                    full_path = os.path.join(root, file)
                    all_issues.extend(analyze_file(full_path))
    return all_issues

# -----------------------
# Main CLI
# -----------------------
if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: python py311_readiness_analyzer.py <file_or_dir>")
        sys.exit(1)

    path = sys.argv[1]
    issues = analyze_path(path)

    if issues:
        print(f"Python 3.11 readiness issues found ({len(issues)}):")
        for issue in issues:
            print("  -", issue)
        sys.exit(1)
    else:
        print(f"No obvious Python 3.11 issues detected in {path} (static check).")

✅ What this full analyzer detects

  1. Unsafe None comparisons (<, >, etc.)

  2. Heuristic detection of date vs datetime arithmetic

  3. Pandas .dt accessor usage without explicit datetime conversion

  4. Ambiguous truthiness on common data variables (df, series, array)

  5. Use of print, input, or open that might trigger string/bytes issues

  6. Can be extended for:

    • is vs == for strings

    • Exact exception message assertions in tests

    • Deprecated module/function usage

Usage

# Analyze a single file
python py311_readiness_analyzer.py my_script.py

# Analyze a whole project
python py311_readiness_analyzer.py ./my_project/
